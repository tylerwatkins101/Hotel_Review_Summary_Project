import hotel_review_summarizer
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
sns.set_style("white")
%matplotlib inline

import warnings
warnings.filterwarnings('ignore')
warnings.simplefilter('ignore')

pd.set_option('display.float_format', lambda x: '{:.3f}'.format(x)) #Limiting floats output to 3 decimal points

data = pd.read_csv('cleaned_reviews.csv')



def full_hotel_review(hotel_name):
    
    """Prints hotel name and calls the other hotel rating summary functions."""
    
    print(hotel_name)
    hotel = data[data.name == hotel_name]
    hotel_review_summarizer.rating_reviews(hotel)
    plt.figure
    hotel_review_summarizer.generate_wordcloud(hotel)
    plt.figure
    hotel_review_summarizer.get_sentiment(hotel)
    hotel_review_summarizer.get_topics(hotel)
